    let currentIdx = 0, score = 0, studentName = "", userChoices = [];
    let attempts = parseInt(localStorage.getItem("nafes_math_attempts_final") || "0");
    document.getElementById("attempts-info").innerText = `محاولاتكِ المستخدمة: ${attempts} من 3`;

    if (attempts >= 3) {
        document.getElementById("start-screen").innerHTML = "<p style='color:red; font-weight:bold;'>لقد استنفدتِ المحاولات الثلاث المتاحة لهذا الجهاز.</p>";
    }

    function startQuiz() {
        studentName = document.getElementById("student-name").value.trim();
        if (studentName.length < 5) { return alert("فضلاً أدخلي اسمكِ الثلاثي للبدء"); }
        document.getElementById("start-screen").classList.add("hidden");
        document.getElementById("quiz-screen").classList.remove("hidden");
        showQuestion();
    }

    function showQuestion() {
        const q = questions[currentIdx];
        document.getElementById("progress").innerText = `السؤال ${currentIdx + 1} من 20`;
        document.getElementById("question-text").innerText = q.q;
        const container = document.getElementById("options-container");
        container.innerHTML = "";
        q.options.forEach((opt, i) => {
            const btn = document.createElement("button");
            btn.className = "option-btn";
            btn.innerText = opt;
            btn.onclick = () => { 
                userChoices.push({q: q.q, selected: opt, correct: q.options[q.correct], isCorrect: i === q.correct}); 
                if(i === q.correct) score++; 
                currentIdx++; 
                if(currentIdx < questions.length) showQuestion(); else finishQuiz(); 
            };
            container.appendChild(btn);
        });
    }

    function finishQuiz() {
        document.getElementById("quiz-screen").classList.add("hidden");
        document.getElementById("result-screen").classList.remove("hidden");
        document.getElementById("score-display").innerText = `درجتكِ هي: ${score} من 20`;
        
        attempts++;
        localStorage.setItem("nafes_math_attempts_final", attempts);

        fetch(SCRIPT_URL, { 
            method: "POST", 
            mode: "no-cors", 
            headers: {"Content-Type": "application/json"},
            body: JSON.stringify({studentName: studentName, score: score}) 
        }).then(() => { 
            document.getElementById("loading-msg").innerText = "✅ تم حفظ نتيجتك في سجلات أ. شهرة بنجاح!";
            showReview();
        }).catch(() => {
            document.getElementById("loading-msg").innerText = "⚠️ لم يتم الاتصال بالسجل، لكن درجتكِ ظهرت أمامكِ.";
            showReview();
        });
    }

    function showReview() {
        let reviewHTML = "<hr><h3>مراجعة شاملة لإجاباتك:</h3>";
        userChoices.forEach((item, idx) => {
            reviewHTML += `<div class="review-box">
                <strong>س${idx+1}:</strong> ${item.q}<br>
                <span class="${item.isCorrect ? 'correct-text' : 'wrong-text'}">
                    إجابتك: ${item.selected} ${item.isCorrect ? '✅' : '❌'}
                </span>
                ${!item.isCorrect ? `<br><small style="color:#555">الإجابة الصحيحة: ${item.correct}</small>` : ''}
            </div>`;
        });
        document.getElementById("review-list").innerHTML = reviewHTML;
    }
